\block{1. The Problem Context}{
    \textbf{Context:} Modern warehouse fleets require dynamic Multi-Robot Task Allocation (MRTA) to operate efficiently. \\[4pt]
    \textbf{The Flaw:} Consensus algorithms (CBBA, GCBBA) assume global connectivity or static links. \\[4pt]
    \textbf{The Reality:} Range-limited radios, obstacles, and agent mobility produce dynamic partitions; baselines can fail to reach consensus and stall operation.
}

\block{2. The Solution: LCBA}{
    The \textbf{Local Consensus-Based Bundle Algorithm (LCBA)} relaxes the global connectivity assumption.
    \begin{itemize}
        \item \textbf{Component-local consensus:} Bidding and consensus operate within each current connected component.
        \item \textbf{Dynamic resolution:} Time-stamped allocations resolve conflicts on reconnection.
        \item \textbf{Event-driven replanning:} Supports continuous task arrivals without full a priori knowledge.
    \end{itemize}
}

\block{8. Conclusion \& Future Work}{
    \textbf{Conclusion:} LCBA successfully bridges the gap between mathematically rigorous task allocation and the messy reality of physical robot deployments. By localizing consensus, fleets can operate continuously despite extreme network fragmentation. \\[4pt]
    \textbf{Future Work:} Integrating fully decentralized collision avoidance and deploying the framework onto physical warehouse robots.

\block{5. Batch Regime: Communication Robustness}{
    \begin{center}
        \includegraphics[width=0.9\linewidth, height=15cm, keepaspectratio]{figures/gridworld_kiva/20260425_030403/batch_results/compact_views/compact_heatmap_completion_rate.png}
        \vspace{0.1cm}
        {\textbf{\\Figure:} Batch completion rate heatmap across algorithms \\(Kiva, 36$\times$36, 20 agents).}
    \end{center}
    \vspace{0.1cm}
    \begin{itemize}
        \item \textbf{Robustness:} LCBA maintains $\geq$0.90 completion across all comm ranges and task loads; CBBA is similarly robust --- both handle network fragmentation gracefully.
        \item \textbf{SOTA Failure:} DMCHBA (SOTA) drops to 32--43\% completion at minimum range --- its per-component Hungarian solves multiply in cost as the network fragments.
        \item \textbf{Clean signal:} No missing data (no wall-clock timeouts) --- every cell reflects a completed run, making the contrast unambiguous.
    \end{itemize}
}

\block{6. Steady-State Regime: Throughput Under Fragmentation}{
    \begin{center}
        \includegraphics[width=0.9\linewidth, height=15cm, keepaspectratio]{figures/gridworld_warehouse_small/20260426_135556/ss_results/compact_views/compact_heatmap_throughput.png}
        \vspace{0.1cm}
        {\textbf{\\Figure:} Steady-state throughput heatmap across algorithms \\(Warehouse Small, 30$\times$30, 6 agents).}
    \end{center}
    \vspace{0.1cm}
    \begin{itemize}
        \item \textbf{Robustness:} LCBA maintains viable throughput across all arrival rates and comm ranges --- no stalls, no missing cells, even at minimum range (4 grid units).
        \item \textbf{SOTA outperformed:} At minimum comm range, LCBA achieves $+18\%$ higher throughput than CBBA and $+66\%$ higher than DMCHBA (SOTA).
        \item \textbf{Convergence:} At full connectivity, all methods reach the same throughput ceiling --- LCBA imposes no overhead when communication is unrestricted.
    \end{itemize}
}

\block{7. Allocation Efficiency: Why LCBA Outperforms CBBA}{
    \begin{center}
        \includegraphics[width=0.95\linewidth, height=15cm, keepaspectratio]{figures/poster_alloc_time_wh_large.png}
        \vspace{0.1cm}
        {\textbf{\\Figure:} Mean allocation time vs.\ communication range (Warehouse Large, 60$\times$60, 18 agents).}
    \end{center}
    \vspace{0.1cm}
    \begin{itemize}
        \item \textbf{LCBA vs.\ CBBA:} At minimum comm range, LCBA is $\mathbf{8\times}$ faster than CBBA --- local consensus terminates early within each component; CBBA wastes all $N_{\min}\times D$ rounds regardless.
        \item \textbf{LCBA vs.\ DMCHBA:} LCBA is $\mathbf{5\times}$ faster than DMCHBA --- DMCHBA clones the full task list and runs Hungarian independently per component, multiplying overhead.
        \item \textbf{Scales with fleet size:} The gap widens as agents increase --- at 18 agents, CBBA's wasted consensus rounds compound, while LCBA's early termination scales favorably.
    \end{itemize}
}
